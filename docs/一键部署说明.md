# RuoYi-Flowable-Plus 一键部署说明

> **目标服务器**：`192.168.1.217`
> **前提条件**：服务器已安装 JDK 1.8、MySQL 8.0、Redis 6.2、Nginx

---

## 一、数据库初始化

### 1.1 创建数据库

```bash
mysql -uroot -proot -e "CREATE DATABASE IF NOT EXISTS \`ry-flowable-plus\` DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;"
```

### 1.2 导入 SQL 脚本

SQL 脚本位于本地 `D:\workspace\sql` 目录，所有文件均以 `ry-flowable-plus_` 为前缀。

**方式一：本地远程导入（推荐）**

在本地 Windows PowerShell 执行：

```powershell
# 获取所有 SQL 文件并按文件名排序后依次导入
Get-ChildItem "D:\workspace\sql\ry-flowable-plus_*.sql" | Sort-Object Name | ForEach-Object {
    Write-Host "正在导入: $($_.Name)" -ForegroundColor Green
    mysql -h 192.168.1.217 -uroot -proot ry-flowable-plus < $_.FullName
}
```

**方式二：上传后导入**

```powershell
# 1. 上传 SQL 文件到服务器
scp D:\workspace\sql\ry-flowable-plus_*.sql root@192.168.1.217:/tmp/sql/

# 2. SSH 登录服务器后批量导入
ssh root@192.168.1.217
cd /tmp/sql
for f in ry-flowable-plus_*.sql; do echo "导入: $f"; mysql -uroot -proot ry-flowable-plus < "$f"; done
```

> **说明**：SQL 文件包含系统表（`sys_*`）、Flowable 流程表（`act_*`、`flw_*`）、工作流表（`wf_*`）、预算模块表（`budget_*`）、代码生成表（`gen_*`）及菜单数据等，全部导入即可。

## 五、配置 Nginx

### Linux 版本

在服务器上创建 `/ruoyi/nginx/conf/nginx.conf`：

```nginx
worker_processes  1;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;

events {
    worker_connections  1024;
}

http {
    include       mime.types;
    default_type  application/octet-stream;
    sendfile        on;
    keepalive_timeout  65;
    client_max_body_size 100m;

    gzip on;
    gzip_min_length 1k;
    gzip_comp_level 6;
    gzip_types text/plain application/javascript application/x-javascript text/css application/xml text/javascript;

    upstream server {
        server 127.0.0.1:8080;
    }

    server {
        listen       80;
        server_name  localhost;

        # 拦截 actuator 端点
        location ~ ^(/[^/]*)?/actuator(/.*)?$ {
            return 403;
        }

        # 前端页面
        location / {
            root   /usr/share/nginx/html;
            try_files $uri $uri/ /index.html;
            index  index.html index.htm;
        }

        # 后端 API 反向代理
        location /prod-api/ {
            proxy_set_header Host $http_host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header REMOTE-HOST $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_pass http://server/;
        }
    }
}
```

## 六、创建管理脚本

### Linux 版本

在服务器 `/ruoyi/server/` 下创建 `app.sh`：

```bash
#!/bin/bash
APP_DIR="/ruoyi/server"
APP_JAR="$APP_DIR/app.jar"
APP_PID="$APP_DIR/app.pid"
LOG_FILE="$APP_DIR/logs/startup.log"
GREEN='\033[0;32m'
RED='\033[0;31m'
NC='\033[0m'

stop_app() {
    if [ -f "$APP_PID" ]; then
        PID=$(cat "$APP_PID")
        if kill -0 "$PID" 2>/dev/null; then
            echo -e "${RED}停止服务 (PID: $PID)...${NC}"
            kill "$PID"
            sleep 3
        fi
        rm -f "$APP_PID"
    fi
}

start_app() {
    if [ -f "$APP_PID" ] && kill -0 "$(cat $APP_PID)" 2>/dev/null; then
        echo -e "${RED}服务已在运行 (PID: $(cat $APP_PID))${NC}"
        return 1
    fi
    echo -e "${GREEN}启动 RuoYi-Flowable-Plus...${NC}"
    cd "$APP_DIR"
    nohup java -Xms512m -Xmx1024m \
         -Djava.security.egd=file:/dev/./urandom \
         -Dserver.port=8080 \
         -Dspring.config.additional-location=$APP_DIR/application-dev.yml \
         --add-opens java.base/java.lang=ALL-UNNAMED \
         --add-opens java.base/java.lang.reflect=ALL-UNNAMED \
         --add-opens java.base/java.util=ALL-UNNAMED \
         -jar app.jar \
         > "$LOG_FILE" 2>&1 &
    echo $! > "$APP_PID"
    echo -e "${GREEN}服务已启动 (PID: $(cat $APP_PID))${NC}"
}

status_app() {
    if [ -f "$APP_PID" ] && kill -0 "$(cat $APP_PID)" 2>/dev/null; then
        echo -e "${GREEN}服务运行中 (PID: $(cat $APP_PID))${NC}"
    else
        echo -e "${RED}服务未启动${NC}"
    fi
}

case "$1" in
    start)   start_app ;;
    stop)    stop_app ;;
    restart) stop_app; sleep 2; start_app ;;
    status)  status_app ;;
    log)     tail -f "$LOG_FILE" ;;
    *)       echo "用法: $0 {start|stop|restart|status|log}" ;;
esac
```

```bash
chmod +x /ruoyi/server/app.sh
```

### Windows 版本

在服务器 `D:\ruoyi\server\` 下创建 `app.ps1`：

```powershell
param(
    [Parameter(Position=0)]
    [ValidateSet('start','stop','restart','status','log')]
    [string]$Action
)

$APP_DIR   = 'D:\workspace\ruoyi-demo\server'
$APP_JAR   = "$APP_DIR\app.jar"
$PID_FILE  = "$APP_DIR\app.pid"
$LOG_FILE  = "$APP_DIR\logs\startup.log"
$JAVA_OPTS = '-Xms512m -Xmx1024m -Djava.security.egd=file:/dev/./urandom -Dserver.port=8080 -Dspring.config.additional-location=D:\workspace\ruoyi-demo\server\application-dev.yml --add-opens java.base/java.lang=ALL-UNNAMED --add-opens java.base/java.lang.reflect=ALL-UNNAMED --add-opens java.base/java.util=ALL-UNNAMED'

function Start-App {
    if (Test-Path $PID_FILE) {
        $procId = Get-Content $PID_FILE
        try {
            $proc = Get-Process -Id $procId -ErrorAction Stop
            Write-Host "Already running (PID: $procId)" -ForegroundColor Red
            return
        } catch {}
    }
    Write-Host 'Starting RuoYi-Flowable-Plus...' -ForegroundColor Green
    $startInfo = @{
        FilePath               = 'java'
        ArgumentList           = "$JAVA_OPTS -jar $APP_JAR"
        WorkingDirectory       = $APP_DIR
        RedirectStandardOutput = $LOG_FILE
        RedirectStandardError  = "$APP_DIR\logs\startup-error.log"
        WindowStyle            = 'Hidden'
        PassThru               = $true
    }
    $process = Start-Process @startInfo
    $process.Id | Out-File -FilePath $PID_FILE -Encoding ASCII
    Write-Host "Started (PID: $($process.Id))" -ForegroundColor Green
}

function Stop-App {
    if (Test-Path $PID_FILE) {
        $procId = Get-Content $PID_FILE
        try {
            $proc = Get-Process -Id $procId -ErrorAction Stop
            Write-Host "Stopping (PID: $procId)..." -ForegroundColor Red
            Stop-Process -Id $procId
            Start-Sleep -Seconds 3
        } catch {
            Write-Host 'Process not found' -ForegroundColor Yellow
        }
        Remove-Item $PID_FILE -Force
    }
}

function Status-App {
    if (Test-Path $PID_FILE) {
        $procId = Get-Content $PID_FILE
        try {
            Get-Process -Id $procId -ErrorAction Stop | Out-Null
            Write-Host "Running (PID: $procId)" -ForegroundColor Green
            return
        } catch {}
    }
    Write-Host 'Not running' -ForegroundColor Red
}

switch ($Action) {
    'start'   { Start-App }
    'stop'    { Stop-App }
    'restart' { Stop-App; Start-Sleep 2; Start-App }
    'status'  { Status-App }
    'log'     { Get-Content $LOG_FILE -Wait -Tail 50 }
    default   { Write-Host 'Usage: .\app.ps1 {start|stop|restart|status|log}' }
}
```

## 十一、常见问题

| 问题 | Linux 排查方法 | Windows 排查方法 |
|------|--------------|----------------|
| 页面打不开 | `systemctl status nginx` | 任务管理器检查 Nginx 进程；`cd D:\nginx-1.30.3\nginx-1.30.3` 执行 `start nginx` |
| 接口 502 | `/ruoyi/server/app.sh status`；`tail -200 /ruoyi/server/logs/startup.log` | `D:\ruoyi\server\app.ps1 status`；`Get-Content D:\ruoyi\server\logs\startup.log -Tail 200` |
| 数据库连接失败 | 确认 MySQL 已启动，检查 `application-dev.yml` 中地址和密码 | 同左 |
| Redis 连接失败 | `redis-cli ping` 测试 Redis 是否正常 | 同左 |
| 端口被占用 | `lsof -i:8080` 或 `lsof -i:80` | `netstat -ano \| findstr :8080` |
| 前端页面刷新 404 | 检查 Nginx 配置中 `try_files` 是否包含 `/index.html` | 同左 |
